# =============================================================================
# Native IDE Tools CMake Configuration
# 100+ Tools for IDE Integration — Zero External Dependencies
# =============================================================================

cmake_minimum_required(VERSION 3.16)

# Source files
set(NATIVE_IDE_TOOLS_HEADERS
    native_ide_tools.h
)

set(NATIVE_IDE_TOOLS_SOURCES
    native_ide_tools.cpp
    native_ide_tools_extended.cpp
)

# Create the library
add_library(native_ide_tools STATIC
    ${NATIVE_IDE_TOOLS_HEADERS}
    ${NATIVE_IDE_TOOLS_SOURCES}
)

# Set C++17 standard
target_compile_features(native_ide_tools PUBLIC cxx_std_17)

# Include directories
target_include_directories(native_ide_tools PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_SOURCE_DIR}/src
)

# Platform-specific libraries
if(WIN32)
    target_link_libraries(native_ide_tools PRIVATE
        kernel32
        user32
        shell32
        ole32
        oleaut32
    )
    
    # Windows-specific definitions
    target_compile_definitions(native_ide_tools PUBLIC
        WIN32_LEAN_AND_MEAN
        UNICODE
        _UNICODE
        _CRT_SECURE_NO_WARNINGS
    )
else()
    target_link_libraries(native_ide_tools PRIVATE
        pthread
        dl
        rt
    )
    
    # POSIX-specific definitions
    target_compile_definitions(native_ide_tools PUBLIC
        _GNU_SOURCE
        _XOPEN_SOURCE=600
    )
endif()

# Compiler-specific options
if(MSVC)
    target_compile_options(native_ide_tools PRIVATE
        /W4
        /MP
        /EHsc
    )
    
    # Enable PDB generation for debugging
    target_compile_options(native_ide_tools PRIVATE
        $<$<CONFIG:Debug>:/Zi>
        $<$<CONFIG:Release>:/Zi>
    )
    
    # Link-time optimization for release
    target_compile_options(native_ide_tools PRIVATE
        $<$<CONFIG:Release>:/GL>
    )
    target_link_options(native_ide_tools PRIVATE
        $<$<CONFIG:Release>:/LTCG>
    )
elseif(CMAKE_CXX_COMPILER_ID MATCHES "GNU" OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    target_compile_options(native_ide_tools PRIVATE
        -Wall
        -Wextra
        -Wpedantic
        -Wno-unused-function
        -fPIC
    )
    
    # Link-time optimization for release
    target_compile_options(native_ide_tools PRIVATE
        $<$<CONFIG:Release>:-flto>
    )
    target_link_options(native_ide_tools PRIVATE
        $<$<CONFIG:Release>:-flto>
    )
endif()

# Integration with existing RawrXD components
# Link against hotpatch_playback for model analysis
target_link_libraries(native_ide_tools PUBLIC
    hotpatch_playback
)

# Optional: Link against inference components if available
if(TARGET inference_playback_bridge)
    target_link_libraries(native_ide_tools PUBLIC
        inference_playback_bridge
    )
    target_compile_definitions(native_ide_tools PUBLIC
        RAWRXD_INFERENCE_BRIDGE_ENABLED
    )
endif()

# Optional: Link against agentic executor if available
if(TARGET agentic_executor)
    target_link_libraries(native_ide_tools PUBLIC
        agentic_executor
    )
    target_compile_definitions(native_ide_tools PUBLIC
        RAWRXD_AGENTIC_EXECUTOR_ENABLED
    )
endif()

# Test executable
option(NATIVE_IDE_TOOLS_BUILD_TESTS "Build test suite" ON)

if(NATIVE_IDE_TOOLS_BUILD_TESTS)
    # Test source
    set(NATIVE_IDE_TOOLS_TEST_SOURCES
        native_ide_tools_test.cpp
    )
    
    add_executable(native_ide_tools_test
        ${NATIVE_IDE_TOOLS_TEST_SOURCES}
    )
    
    target_link_libraries(native_ide_tools_test PRIVATE
        native_ide_tools
    )
    
    enable_testing()
    add_test(NAME NativeIDEToolsTest COMMAND native_ide_tools_test)
endif()

# Demo executable for smoke testing
option(NATIVE_IDE_TOOLS_BUILD_DEMO "Build demo programs" ON)

if(NATIVE_IDE_TOOLS_BUILD_DEMO)
    add_executable(native_ide_tools_demo
        native_ide_tools_demo.cpp
    )
    
    target_link_libraries(native_ide_tools_demo PRIVATE
        native_ide_tools
    )
    
    # Demo definitions
    target_compile_definitions(native_ide_tools_demo PRIVATE
        NATIVE_IDE_TOOLS_DEMO
    )
endif()

# Installation
install(TARGETS native_ide_tools
    ARCHIVE DESTINATION lib
    LIBRARY DESTINATION lib
)

install(FILES ${NATIVE_IDE_TOOLS_HEADERS}
    DESTINATION include/rawrxd/core
)

# Export targets for use by other projects
export(TARGETS native_ide_tools
    FILE "${CMAKE_CURRENT_BINARY_DIR}/native_ide_tools_targets.cmake"
)

# Package configuration
include(CMakePackageConfigHelpers)

configure_package_config_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/native_ide_tools_config.cmake.in"
    "${CMAKE_CURRENT_BINARY_DIR}/native_ide_tools_config.cmake"
    INSTALL_DESTINATION lib/cmake/native_ide_tools
)

write_basic_package_version_file(
    "${CMAKE_CURRENT_BINARY_DIR}/native_ide_tools_version.cmake"
    VERSION 1.0.0
    COMPATIBILITY SameMajorVersion
)

install(FILES
    "${CMAKE_CURRENT_BINARY_DIR}/native_ide_tools_config.cmake"
    "${CMAKE_CURRENT_BINARY_DIR}/native_ide_tools_version.cmake"
    DESTINATION lib/cmake/native_ide_tools
)
