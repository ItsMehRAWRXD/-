#include "Detour.hpp"
#include <windows.h>
#include <cstring>

namespace RawrXD::Agentic::Hotpatch {

extern "C" void RawrXD_GenerateAbsoluteJump_MASM(void* buf, void* target);
extern "C" bool RawrXD_GenerateRelativeJump_MASM(void* from, void* to, void* buf);

void Detour::generateAbsoluteJump(void* from, void* to, uint8_t* buffer) {
    RawrXD_GenerateAbsoluteJump_MASM(buffer, to);
}

bool Detour::generateRelativeJump(void* from, void* to, uint8_t* buffer) {
    return RawrXD_GenerateRelativeJump_MASM(from, to, buffer);
}

void* Detour::allocateTrampolineNear(void* target, size_t size) {
    return VirtualAlloc(nullptr, size, MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
}

void Detour::freeTrampoline(void* trampoline) {
    VirtualFree(trampoline, 0, MEM_RELEASE);
}

void* Detour::install(void* targetFunc, void* replacementFunc, size_t* trampolineSize) {
    size_t requiredBytes = calculateRequiredBytes(targetFunc);
    void* trampoline = allocateTrampolineNear(targetFunc, requiredBytes + 32);
    if (!trampoline) return nullptr;

    DWORD oldProtect;
    if (!VirtualProtect(targetFunc, requiredBytes, PAGE_EXECUTE_READWRITE, &oldProtect)) {
        freeTrampoline(trampoline);
        return nullptr;
    }

    memcpy(trampoline, targetFunc, requiredBytes);
    uint8_t* trampolineEnd = (uint8_t*)trampoline + requiredBytes;
    generateAbsoluteJump(trampolineEnd, (uint8_t*)targetFunc + requiredBytes, trampolineEnd);

    // Patch target function with a jump to replacement
    generateAbsoluteJump(targetFunc, replacementFunc, (uint8_t*)targetFunc);

    VirtualProtect(targetFunc, requiredBytes, oldProtect, &oldProtect);
    FlushInstructionCache(GetCurrentProcess(), targetFunc, requiredBytes);

    if (trampolineSize) *trampolineSize = requiredBytes + 14;
    return trampoline;
}

bool Detour::uninstall(void* targetFunc, void* trampoline) {
    if (!targetFunc || !trampoline) return false;

    // Size is saved in trampoline logic, but for uninstall we assume 14 bytes jump overwrote target
    size_t sz = 14;

    DWORD old;
    if (!VirtualProtect(targetFunc, sz, PAGE_EXECUTE_READWRITE, &old)) return false;
    memcpy(targetFunc, trampoline, sz);
    VirtualProtect(targetFunc, sz, old, &old);

    FlushInstructionCache(GetCurrentProcess(), targetFunc, sz);
    freeTrampoline(trampoline);
    return true;
}

size_t Detour::calculateRequiredBytes(void* targetFunc) {
    const uint8_t* code = (const uint8_t*)targetFunc;
    size_t total = 0;
    while (total < 14) {
        InstructionInfo info = disassembleOne(code + total);
        if (info.length == 0) break;
        total += info.length;
    }
    return total;
}

InstructionInfo Detour::disassembleOne(const uint8_t* code) {
    InstructionInfo info{};
    size_t offset = 0;

    // Skip prefixes
    while (true) {
        uint8_t b = code[offset];
        if ((b & 0xF0) == 0x40) { offset++; continue; } // REX
        if (b == 0x66 || b == 0x67 || b == 0xF0 || b == 0xF2 || b == 0xF3) { offset++; continue; }
        break;
    }

    uint8_t op = code[offset++];
    if (op == 0x0F) {
        uint8_t op2 = code[offset++];
        if (op2 == 0x1F || op2 == 0x0D) {
            uint8_t modrm = code[offset++];
            info.length = (uint8_t)offset;
        } else if ((op2 & 0xF0) == 0x80) { // Jcc rel32
            info.length = (uint8_t)(offset + 4);
        } else { info.length = (uint8_t)(offset + 1); }
    } else {
        switch (op) {
            case 0xE8: case 0xE9: info.length = (uint8_t)(offset + 4); break;
            case 0xEB: info.length = (uint8_t)(offset + 1); break;
            case 0x50: case 0x51: case 0x52: case 0x53: case 0x54: case 0x55: case 0x56: case 0x57:
            case 0x58: case 0x59: case 0x5A: case 0x5B: case 0x5C: case 0x5D: case 0x5E: case 0x5F:
            case 0x90: case 0xC3: case 0xCC: info.length = (uint8_t)offset; break;
            case 0x81: case 0x83: case 0x89: case 0x8B: case 0x31: case 0x33: case 0xFF: {
                uint8_t modrm = code[offset++];
                uint8_t mod = (modrm >> 6) & 0x03;
                uint8_t rm = modrm & 0x07;
                size_t len = offset;
                if (mod != 3 && rm == 4) len++; // SIB
                if (mod == 1) len += 1; else if (mod == 2) len += 4; else if (mod == 0 && rm == 5) len += 4;
                if (op == 0x81) len += 4; else if (op == 0x83) len += 1;
                info.length = (uint8_t)len;
                break;
            }
            default: info.length = (uint8_t)offset; if (info.length == 0) info.length = 1; break;
        }
    }
    if (info.length == 0) info.length = 1;
    return info;
}

bool Detour::isNearJumpPossible(void* from, void* to) {
    int64_t diff = (int64_t)to - (int64_t)from;
    return (diff >= -2147483648LL && diff <= 2147483647LL);
}

void* Detour::findCodeCave(void* near, size_t size, size_t searchRadius) {
    uintptr_t base = (uintptr_t)near;
    uintptr_t start = (base > searchRadius) ? base - searchRadius : 0;
    uintptr_t end = base + searchRadius;
    MEMORY_BASIC_INFORMATION mbi;
    for (uintptr_t i = start; i < end; ) {
        if (!VirtualQuery((void*)i, &mbi, sizeof(mbi))) break;
        if (mbi.State == MEM_COMMIT && (mbi.Protect & (PAGE_EXECUTE_READ | PAGE_EXECUTE_READWRITE))) {
            uint8_t* p = (uint8_t*)mbi.BaseAddress;
            size_t count = 0;
            for (size_t k = 0; k < mbi.RegionSize; k++) {
                if (p[k] == 0xCC || p[k] == 0x90) {
                    if (++count >= size) return (void*)(&p[k - size + 1]);
                } else count = 0;
            }
        }
        i = (uintptr_t)mbi.BaseAddress + mbi.RegionSize;
    }
    return nullptr;
}

bool Detour::relocateInstruction(const uint8_t* src, uint8_t* dst, uintptr_t base, size_t* len) {
    *len = (size_t)disassembleOne(src).length;
    memcpy(dst, src, *len);
    return true;
}
